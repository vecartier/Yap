**Step B — Update DetailRouter.swift**

Replace the body with routing based on `selectedSessionID`:

```swift
private enum Content { case live, past(String), empty }

private var resolvedContent: Content {
    if selectedSessionID == "_live_" { return .live }
    if let id = selectedSessionID { return .past(id) }
    return .empty
}

var body: some View {
    switch resolvedContent {
    case .live:
        LiveDetailView(settings: settings)
    case .past(let id):
        PastMeetingDetailView(sessionID: id, settings: settings)
    case .empty:
        if coordinator.sessionHistory.isEmpty {
            // onboarding empty state (keep existing design from current DetailRouter)
            VStack(spacing: 20) { ... }
        } else {
            ContentUnavailableView("Select a Meeting", systemImage: "waveform",
                description: Text("Choose a meeting from the sidebar."))
        }
    }
}
```

DO NOT check `coordinator.isRecording` in resolvedContent. The sentinel `"_live_"` is the sole routing signal — this ensures the "Finalizing..." state stays on LiveDetailView.

**Step D — Update MainAppView.swift**

Add two onChange modifiers to the NavigationSplitView (or its `.navigationSplitViewStyle` chain):

```swift
.onChange(of: coordinator.isRecording) { _, isRecording in
    if isRecording {
        selectedSessionID = "_live_"
    }
    // DO NOT clear on false — that would flash to empty during finalization
}
.onChange(of: coordinator.lastEndedSession) { _, session in
    guard let session else { return }
    selectedSessionID = session.id
}
```

Note: `lastEndedSession` is set at the end of `finalizeCurrentSession()` in AppCoordinator, after `loadHistory()` completes. The onChange fires reliably on the main thread via @Observable.
